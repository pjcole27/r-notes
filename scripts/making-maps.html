# make locations geographic
locations_sf <- st_as_sf(locations, 
                      coords = c("long", "lat"),
                      crs = 4326)

Project the data

state_plane(state = "VA")
[1] 32146
locations_proj <- st_transform(locations_sf, crs="EPSG:32146")
counties_proj <- st_transform(counties, crs = "EPSG:32146")

Let’s make a map with ggplot

ggplot() +
  geom_sf(data = counties_proj, fill = NA) + 
  geom_sf(data = locations_proj) +
  theme_light()
